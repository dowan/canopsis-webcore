Ember.Application.initializer({
    name: 'CalendardayAdapter',
    after: ['ApplicationAdapter'],
    initialize: function(container, application) {
        var ApplicationAdapter = container.lookupFactory('adapter:application');

        /**
         * @adapter calendarday
         * @memberOf canopsis.frontend.brick-calendar
         */
        var adapter = ApplicationAdapter.extend({

            /**
             * Ember method to query database
             * @method findQuery
             * @param {string} type used for buildURL
             * @param {string} store store to query
             * @param {object} query mongoDB query
             * @return {array} list of events
             */
            findQuery: function(store, type, query) {
                var url = '/eventslog/count';

                return this.ajax(url, 'GET', { data: query });
            }
        });

        application.register('adapter:calendarday', adapter);

    }

});
