Ember.Application.initializer({
    name: 'UserviewsimplemodelAdapter',
    after: 'UserviewAdapter',
    initialize: function(container, application) {
        var UserviewAdapter = container.lookupFactory('adapter:userview');

        /**
         * @adapter userviewsimplemodel
         */
        var adapter = UserviewAdapter.extend({});

        application.register('adapter:userviewsimplemodel', adapter);
    }
});
